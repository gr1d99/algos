def call(result, *nums)
  base_cases_result = base_cases(result, nums)

  return base_cases_result unless base_cases_result.nil?

  can_sum = false

  nums.each do |num|
    can_sum = call(result - num, *nums)

    break if can_sum
  end

  can_sum
end
